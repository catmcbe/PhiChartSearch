import wave
import os

class AudioFile:
    def __init__(self, file_path):
        self.file_path = file_path
        self.file_name = os.path.basename(file_path)
        self.duration = 0
        self.match_score = 0
        
        self.calculate_duration()
    
    def calculate_duration(self):
        """è®¡ç®—éŸ³é¢‘æ—¶é•¿"""
        try:
            with wave.open(self.file_path, 'rb') as wav_file:
                # è·å–éŸ³é¢‘å‚æ•°
                frames = wav_file.getnframes()
                sample_rate = wav_file.getframerate()
                
                # è®¡ç®—æ—¶é•¿ï¼ˆç§’ï¼‰
                self.duration = frames / float(sample_rate)
                
        except Exception as e:
            print(f"è¯»å–éŸ³é¢‘æ–‡ä»¶å¤±è´¥ {self.file_path}: {e}")
            self.duration = 0
    
    def calculate_match_score(self, target_duration):
        """è®¡ç®—åŒ¹é…åº¦"""
        if self.duration <= 0 or target_duration <= 0:
            return 0
        
        # è®¡ç®—æ—¶é—´å·®
        time_diff = abs(self.duration - target_duration)
        
        # åŒ¹é…åº¦è®¡ç®—ï¼šæ—¶é—´å·®è¶Šå°ï¼ŒåŒ¹é…åº¦è¶Šé«˜
        # ä½¿ç”¨ç®€å•çš„çº¿æ€§å…¬å¼
        if time_diff <= 1.0:  # 1ç§’å†…å·®å¼‚
            self.match_score = 100 - (time_diff * 50)
        else:
            self.match_score = max(0, 50 - (time_diff * 10))
        
        return self.match_score

def find_matching_audio_files(audio_folder, target_duration, min_score=50):
    """æŸ¥æ‰¾åŒ¹é…çš„éŸ³é¢‘æ–‡ä»¶"""
    matching_files = []
    
    if not os.path.exists(audio_folder):
        return matching_files
    
    for file_name in os.listdir(audio_folder):
        if file_name.lower().endswith('.wav'):
            file_path = os.path.join(audio_folder, file_name)
            audio_file = AudioFile(file_path)
            
            # è®¡ç®—åŒ¹é…åº¦
            score = audio_file.calculate_match_score(target_duration)
            
            if score >= min_score:
                matching_files.append(audio_file)
    
    # æŒ‰åŒ¹é…åº¦æ’åº
    matching_files.sort(key=lambda x: x.match_score, reverse=True)
    
    return matching_files
